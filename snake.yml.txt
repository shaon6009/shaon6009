name: Generate Snake Animation

on:
  schedule: # execute every 12 hours
    - cron: "0 */12 * * *"  # Correct cron expression for every 12 hours
  workflow_dispatch:  # Allow manual trigger
  push:
    branches:
      - master  # Trigger on push to the master branch

jobs:
  generate:
    permissions:
      contents: write  # Grant permission to write to the repository
    runs-on: ubuntu-latest
    timeout-minutes: 5  # Set a timeout of 5 minutes for the job

    steps:
      - name: Generate snake.svg
        uses: Platane/snk/svg-only@v3
        with:
          github_user_name: ${{ github.repository_owner }}
          outputs: dist/snake.svg?palette=github-dark  # Customize the palette for dark mode

      - name: Push snake.svg to the output branch
        uses: crazy-max/ghaction-github-pages@v3.1.0
        with:
          target_branch: output  # Specify the branch to push to
          build_dir: dist  # Path to the generated file
        env:
          GITHUB_TOKEN: ${{ secrets.GITHUB_TOKEN }}  # Ensure this secret is set in your repo settings
